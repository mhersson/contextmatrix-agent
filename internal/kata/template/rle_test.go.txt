package kata

import "testing"

func TestEncode(t *testing.T) {
	cases := []struct{ in, want string }{
		{"", ""},
		{"a", "1a"},
		{"aaabbc", "3a2b1c"},
		{"aaaa", "4a"},
		{"abc", "1a1b1c"},
	}
	for _, c := range cases {
		if got := Encode(c.in); got != c.want {
			t.Errorf("Encode(%q) = %q, want %q", c.in, got, c.want)
		}
	}
}

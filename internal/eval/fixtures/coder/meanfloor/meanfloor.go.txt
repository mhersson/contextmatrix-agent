package meanfloor

// MeanFloor returns the floor of the exact arithmetic mean of xs, rounding toward
// negative infinity (not toward zero). An empty slice returns 0. There are bugs
// below — fix them so the tests pass.
func MeanFloor(xs []int64) int64 {
	if len(xs) == 0 {
		return 0
	}
	var sum int64
	for _, x := range xs {
		sum += x
	}
	return sum / int64(len(xs))
}

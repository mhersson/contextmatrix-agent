package meanfloor

import (
	"math"
	"testing"
)

func TestMeanFloor(t *testing.T) {
	cases := []struct {
		name string
		in   []int64
		want int64
	}{
		{"empty", nil, 0},
		{"single positive", []int64{5}, 5},
		{"single negative", []int64{-5}, -5},
		{"exact even", []int64{2, 4}, 3},
		{"exact three", []int64{1, 2, 3}, 2},
		{"positive floor", []int64{1, 2}, 1},
		{"positive floor four", []int64{5, 5, 5, 1}, 4},
		{"negative floor half", []int64{-1, -2}, -2},
		{"negative floor third", []int64{-3, -3, -2}, -3},
		{"mixed floor neg half", []int64{1, -2}, -1},
		{"overflow max three", []int64{math.MaxInt64, math.MaxInt64, math.MaxInt64}, math.MaxInt64},
		{"overflow max two", []int64{math.MaxInt64, math.MaxInt64}, math.MaxInt64},
		{"overflow min two", []int64{math.MinInt64, math.MinInt64}, math.MinInt64},
		{"max plus min", []int64{math.MaxInt64, math.MinInt64}, -1},
		{"two mins plus one overflow", []int64{math.MinInt64, math.MinInt64, 1}, -6148914691236517205},
	}
	for _, c := range cases {
		t.Run(c.name, func(t *testing.T) {
			if got := MeanFloor(c.in); got != c.want {
				t.Errorf("MeanFloor(%v) = %d, want %d", c.in, got, c.want)
			}
		})
	}
}

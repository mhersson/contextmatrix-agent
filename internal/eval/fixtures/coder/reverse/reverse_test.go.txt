package reverse

import "testing"

func TestReverse(t *testing.T) {
	cases := []struct{ in, want string }{
		{"", ""},
		{"abc", "cba"},
		{"Go", "oG"},
		{"racecar", "racecar"},
		{"héllo", "olléh"},
	}
	for _, c := range cases {
		if got := Reverse(c.in); got != c.want {
			t.Errorf("Reverse(%q) = %q, want %q", c.in, got, c.want)
		}
	}
}

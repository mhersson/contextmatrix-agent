package calc

import "testing"

func TestEval(t *testing.T) {
	valid := []struct {
		expr string
		want int
	}{
		{"42", 42},
		{"2+3*4", 14},
		{"(2+3)*4", 20},
		{"2*3+4*5", 26},
		{"10-3-2", 5},
		{"100/10/2", 5},
		{"10/3", 3},
		{"7/2", 3},
		{"1+2+3+4", 10},
		{" 1 + 2 ", 3},
		{"((1+2))", 3},
		{"2*(3+(4*5))", 46},
		{"1000000*1000000", 1000000000000},
	}
	for _, c := range valid {
		t.Run(c.expr, func(t *testing.T) {
			got, err := Eval(c.expr)
			if err != nil {
				t.Errorf("Eval(%q) unexpected error: %v", c.expr, err)
				return
			}
			if got != c.want {
				t.Errorf("Eval(%q) = %d, want %d", c.expr, got, c.want)
			}
		})
	}

	bad := []string{
		"",
		"1+",
		"+1",
		"(1+2",
		"1+2)",
		"1//2",
		"1 2",
		"1/0",
		"6/(3-3)",
		"a+1",
		"1.5",
	}
	for _, expr := range bad {
		t.Run(expr, func(t *testing.T) {
			if _, err := Eval(expr); err == nil {
				t.Errorf("Eval(%q) expected error, got nil", expr)
			}
		})
	}
}

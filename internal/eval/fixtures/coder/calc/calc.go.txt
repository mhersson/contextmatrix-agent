package calc

import (
	"errors"
	"strconv"
)

// Eval evaluates an integer arithmetic expression made of non-negative integer
// literals, the binary operators + - * /, parentheses, and spaces. Operators *
// and / bind tighter than + and -; equal-precedence operators are
// left-associative; division truncates toward zero. It returns an error for
// malformed input or division by zero. There are bugs below — fix them so the
// tests pass.
func Eval(expr string) (int, error) {
	var result, cur int
	op := byte('+')
	haveNum := false
	apply := func() {
		switch op {
		case '+':
			result += cur
		case '-':
			result -= cur
		case '*':
			result *= cur
		case '/':
			if cur != 0 {
				result /= cur
			}
		}
	}
	for i := 0; i < len(expr); {
		c := expr[i]
		switch {
		case c == ' ':
			i++
		case c >= '0' && c <= '9':
			j := i
			for j < len(expr) && expr[j] >= '0' && expr[j] <= '9' {
				j++
			}
			n, err := strconv.Atoi(expr[i:j])
			if err != nil {
				return 0, err
			}
			cur = n
			haveNum = true
			i = j
		case c == '+' || c == '-' || c == '*' || c == '/':
			apply()
			op = c
			cur = 0
			i++
		default:
			return 0, errors.New("invalid character")
		}
	}
	if !haveNum {
		return 0, errors.New("empty expression")
	}
	apply()
	return result, nil
}

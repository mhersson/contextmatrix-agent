package ratelimit

// Limiter is a sliding-window rate limiter. It permits at most capacity events
// within any trailing window of duration window (in abstract time units). Time is
// supplied by the caller via Allow's now argument; the limiter never reads a real
// clock, so its behavior is fully determined by the sequence of timestamps it sees.
type Limiter struct {
	capacity int
	window   int64
	events   []int64 // timestamps of permitted events, oldest first
}

// NewLimiter returns a limiter permitting at most capacity events per trailing
// window time units. A capacity of zero rejects every request.
func NewLimiter(capacity int, window int64) *Limiter {
	return &Limiter{capacity: capacity, window: window}
}

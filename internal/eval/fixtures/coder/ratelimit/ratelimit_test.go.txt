package ratelimit

import "testing"

// step is one Allow call with the timestamp to feed and the verdict expected.
type step struct {
	now  int64
	want bool
}

func run(t *testing.T, l *Limiter, steps []step) {
	t.Helper()

	for i, s := range steps {
		if got := l.Allow(s.now); got != s.want {
			t.Errorf("step %d: Allow(%d) = %v, want %v", i, s.now, got, s.want)
		}
	}
}

// TestBurstThenRejected fills the window to capacity at a single instant, then the
// next request at the same instant is rejected.
func TestBurstThenRejected(t *testing.T) {
	run(t, NewLimiter(3, 10), []step{
		{0, true},  // 1st of 3
		{0, true},  // 2nd of 3
		{0, true},  // 3rd of 3 fills the window
		{0, false}, // 4th overflows
		{0, false}, // still full
	})
}

// TestWindowEdgeExpires checks the trailing-edge boundary: an event recorded at
// time 0 has aged out exactly at now=window (0 <= now-window), but is still live
// one tick earlier.
func TestWindowEdgeExpires(t *testing.T) {
	// Just before the edge: the three events at t=0 are still in window, so the
	// request at now=9 (9-10 = -1, nothing aged out) is rejected.
	run(t, NewLimiter(3, 10), []step{
		{0, true},
		{0, true},
		{0, true},
		{9, false}, // window still holds all three
	})

	// Exactly at the edge: now=10 ages out everything at t<=0, freeing a slot.
	run(t, NewLimiter(3, 10), []step{
		{0, true},
		{0, true},
		{0, true},
		{10, true}, // the three t=0 events have aged out
	})
}

// TestWindowSlides ages events out one at a time as now advances, so a steady
// drip at the window cadence is always permitted.
func TestWindowSlides(t *testing.T) {
	l := NewLimiter(2, 10)
	run(t, l, []step{
		{0, true},   // events: {0}
		{5, true},   // events: {0,5} -> full
		{6, false},  // 0 still live (6-10=-4); two in window -> reject
		{10, true},  // 0 aged out (0<=0); events: {5,10}
		{15, true},  // 5 aged out (5<=5); events: {10,15}
		{16, false}, // 10,15 both live -> reject
	})
}

// TestZeroCapacityAlwaysRejects checks the capacity=0 edge: never permit anything,
// regardless of time.
func TestZeroCapacityAlwaysRejects(t *testing.T) {
	run(t, NewLimiter(0, 10), []step{
		{0, false},
		{100, false},
		{1000, false},
	})
}

// TestCapacityOne is the minimal-throughput sanity check: one event per window,
// with the next admitted only once the previous has aged off the trailing edge.
func TestCapacityOne(t *testing.T) {
	run(t, NewLimiter(1, 5), []step{
		{0, true},  // events: {0}
		{2, false}, // 0 still live (2-5=-3) -> reject
		{4, false}, // 0 still live -> reject
		{5, true},  // 0 aged out (0<=0) -> admit; events: {5}
		{6, false}, // 5 live -> reject
	})
}

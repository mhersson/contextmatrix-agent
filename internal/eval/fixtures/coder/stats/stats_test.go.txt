package stats

import "testing"

func TestMax(t *testing.T) {
	cases := []struct {
		in   []int
		want int
	}{
		{[]int{}, 0},
		{[]int{3, 1, 2}, 3},
		{[]int{-5, -1, -3}, -1},
		{[]int{42}, 42},
	}
	for _, c := range cases {
		if got := Max(c.in); got != c.want {
			t.Errorf("Max(%v) = %d, want %d", c.in, got, c.want)
		}
	}
}

package dedup

import (
	"reflect"
	"testing"
)

func TestDedup(t *testing.T) {
	cases := []struct{ in, want []int }{
		{nil, nil},
		{[]int{1, 1, 2, 3, 3, 3}, []int{1, 2, 3}},
		{[]int{5, 4, 5, 4}, []int{5, 4}},
		{[]int{7}, []int{7}},
	}
	for _, c := range cases {
		if got := Dedup(c.in); !reflect.DeepEqual(got, c.want) {
			t.Errorf("Dedup(%v) = %v, want %v", c.in, got, c.want)
		}
	}
}
